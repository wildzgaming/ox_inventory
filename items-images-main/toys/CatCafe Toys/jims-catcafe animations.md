```
	for _, v in pairs({
					"plushiemini1",
					"plushiemini2",
					"plushiemini3",
					"plushiemini4",
					"plushiemini5",
					"plushiemini6",
					"plushiemini7",
					"plushiemini8",
					"plushiemini9",
					"plushie1",
					"plushie2",
					"plushie3",
					"plushie4",
					"plushie5",
					"plushie6",
					"plushie7",
					"plushie8",
					"plushie9",
				}) do 
			QBCore.Functions.CreateUseableItem(v, function(source, item)
			TriggerClientEvent('jim-catcafe:client:plushieUse', source, item.name)
		end)
	end
```
-- `CLIENT SIDE OF CAT CAFE` -- Add these to the bottom of the file
```
	RegisterNetEvent('jim-catcafe:client:plushieUse', function(item)
		TriggerEvent('animations:client:EmoteCommandStart', {item})
		Wait(10000)
		TriggerEvent('animations:client:EmoteCommandStart', {"c"})
	end)

	RegisterNetEvent('jim-catcafe:client:plushiePick', function()
		local pool = {	"plushiemini1",
						"plushiemini2",
						"plushiemini3",
						"plushiemini4",
						"plushiemini5",
						"plushiemini6",
						"plushiemini7",
						"plushiemini8",
						"plushiemini9", }
		--[[local pool = {  "plushie1",
						"plushie2",
						"plushie3",
						"plushie4",
						"plushie5",
						"plushie6",
						"plushie7",
						"plushie8",
						"plushie9",	} ]]
		if math.random(0,10) == 1 then -- if the random number = 1 then give a prize
			local prize = math.random(1, #pool)
			TriggerEvent("inventory:client:ItemBox", QBCore.Shared.Items[pool[prize]], "add", 1)
			TriggerServerEvent("QBCore:Server:AddItem", pool[prize], 1)
		end
	end)
```
